Code example

My solution for Dublicate Encoder Kata on Codewars 

              
function duplicateEncode(word){
  noCaseWord = word.toLowerCase();
  const lettersArray = noCaseWord.split('');
  let result = ''; 
  for (i = 0; i < lettersArray.length; i++) {
    for (j = 0; j < lettersArray.length; j++) {
      if (i !== j && lettersArray[i] === lettersArray[j]) {
        result += ')';
        break;
      } else if ((j + 1) === lettersArray.length) {
        result += '(';
        break;
      }
    }
  }
  return result;
};
